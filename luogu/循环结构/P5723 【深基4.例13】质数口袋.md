### 暴力

#### 判断质数
```cpp
bool prime(int x){
    if(x<2) return false;
    if(x==2) return true;
    if(x%2==0) return false; // 偶数可以直接排除
    for(int i=3; i*i<=x; i+=2){ // 使用 i * i <= x， 减少遍历次数，同时不用每次都调用 sqrt() 函数
        if(x%i==0) return false;
    }
    return true;;
}
```

#### 筛选 1 ~ 10000 中所有质数并存入数组 a 中
```cpp
for(int i=2; i<=MAX_SIZE; i++){ // 筛选出 0~100000 中所有的质数并存入数组中
        if(prime(i)) a[++top]=i; // 先 ++，因为需要将数组空间扩大之后才能存值
    }
```

#### 计算质数和还有口袋中质数总数
```cpp
int count=0;
    for(int j=1; j<=top; j++){ 
        if(s+a[j]>n) break; // 超过口袋的负载量就跳出循环
        s+=a[j]; // 没有超出就继续累加
        count++;
    }
```

#### 打印结果
```cpp
for(int k=1; k<=count; k++){ // 注意数组是从 1 开始存的所以从 1 开始遍历
        cout<<a[k]<< endl;
    }
    cout<<count;
```

> 注意
> for 循环遍历数组的时候，要看数组从 1 还是 0 开始存。

#### 源代码
```cpp
```cpp
#include<iostream>
#include<cmath>
using namespace std;

const long long MAX_SIZE = 100000;
long long s = 0; // 质数和
long long a[MAX_SIZE]; // 质数数组
long long top = 0; // 数组的最大长度
long long n; // 口袋的负载量

bool prime(int x);

int main(){
    cin>>n;
    for(int i=2; i<=MAX_SIZE; i++){ // 筛选出 0~100000 中所有的质数并存入数组中
        if(prime(i)) a[++top]=i; // 先++，因为需要将数组空间扩大之后才能存值
    }

    int count=0;
    for(int j=1; j<=top; j++){ 
        if(s+a[j]>n) break;
        s+=a[j];
        count++;
    }

    for(int k=1; k<=count; k++){ // 注意数组是从 1 开始存的所以从 1 开始遍历
        cout<<a[k]<< endl;
    }
    cout<<count;
    
    return 0;
}

bool prime(int x){
    if(x<2) return false;
    if(x==2) return true;
    for(int i=2; i<=sqrt(x); i++){
        if(x%i==0) return false;
    }
    return true;
}

```
